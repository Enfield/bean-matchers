1   package com.google.code.beanmatchers;

3   import java.util.List;
4   import org.hamcrest.Description;
5   import org.hamcrest.DiagnosingMatcher;
6   
7   abstract class AbstractBeanAccessorMatcher<T> extends DiagnosingMatcher<T> {
8     final TypeBasedValueGenerator valueGenerator;
9   
10    AbstractBeanAccessorMatcher(TypeBasedValueGenerator valueGenerator) {
11      this.valueGenerator = valueGenerator;
12    }
13  
14    protected boolean beanHasValidGetterAndSetterForProperties(
15        JavaBean bean, List<String> properties, Description mismatchDescription) {
16      for (String property : properties) {
17        if (beanDoesNotHaveValidGetterAndSetterForProperty(bean, property)) {
18          mismatchDescription
19              .appendText("bean of type ")
20              .appendValue(bean.beanType().getName())
21              .appendText(" had an invalid getter/setter for the property ")
22              .appendValue(property);
23          return false;
24        }
25      }
26      return true;
27    }
28  
29    protected boolean beanDoesNotHaveValidGetterAndSetterForProperty(JavaBean bean, String property) {
30      try {
31        Class<?> propertyType = bean.propertyType(property);
32        Object testValue = valueGenerator.generate(propertyType);
33        bean.setProperty(property, testValue);
34        Object result = bean.getProperty(property);
35        return !testValue.equals(result);
36      } catch (AccessorMissingException exception) {
37        return true;
38      }
39    }
40  }
